
# 8、最大二叉树

LeetCode 654. [最大二叉树](https://leetcode-cn.com/problems/maximum-binary-tree/)


```js
var constructMaximumBinaryTree = function (nums) {
  // 划分函数
  function partition(arr) {
    if (arr.length === 0) return [];
    let max = arr[0];
    let maxIndex = 0;
    for (let i = 0; i < arr.length; i++) {
      if (arr[i] > max) {
        max = arr[i];
        maxIndex = i;
      }
    }
    const left = maxIndex === 0 ? [] : arr.slice(0, maxIndex);
    const right = maxIndex === arr.length ? [] : arr.slice(maxIndex + 1);
    return {
      left,
      right,
      max,
    };
  }
  function crateMaxTree(arr) {
    if (!arr || !arr.length) return null;
    if (arr.length === 1) return new TreeNode(arr[0]);
    const { left, right, max } = partition(arr);
    const root = new TreeNode(max);
    // 左右递归
    const l = crateMaxTree(left);
    const r = crateMaxTree(right);
    root.left = l;
    root.right = r;
    return root;
  }
  return crateMaxTree(nums);
};
```
